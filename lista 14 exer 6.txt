 public static void main(String[] args) {

		primeiroNum=entrada.nextInt();

		quant=entrada.nextInt();

	} 
	 
 public static int calcular(int p,int r,int q){
	 	int i;
	 	int res=0;
	 	
		 for(i=1;i<=q;i++){
		
		System.out.println(p);

		 }
		 
		 return res;
	 }
